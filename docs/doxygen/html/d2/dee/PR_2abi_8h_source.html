272 /*
273  * Info about parameters:
274  *
275  * A "count" in the following macros is always measured in bytes.
276  *
277  * All volumes/gains are in Q1.15 signed fixed point numbers:
278  * 0x8000 is the minimum volume (-100%), negating the audio curve.
279  * 0x0000 is silent.
280  * 0x7fff is maximum volume (99.997%).
281  *
282  * All DRAM addresses refer to segmented addresses. A segment table shall
283  * first be set up by calling aSegment for each segment. When a DRAM
284  * address is later used as parameter, the 8 high bits will be an index
285  * to the segment table and the lower 24 bits are added to the base address
286  * stored in the segment table for this entry. The result is the physical address.
287  *
288  * Transfers to/from DRAM are executed using DMA and hence follow these restrictions:
289  * All DRAM addresses should be aligned by 8 bytes, or they will be
290  * rounded down to the nearest multiple of 8 bytes.
291  * All DRAM lengths should be aligned by 8 bytes, or they will be
292  * rounded up to the nearest multiple of 8 bytes.
293  */

295 /*
296  * Decompresses ADPCM data.
297  * Possible flags: A_INIT and A_LOOP.
298  *
299  * First set up internal data in DMEM:
300  * aLoadADPCM(cmd++, nEntries * 16, physicalAddressOfBook)
301  * aSetLoop(cmd++, physicalAddressOfLoopState) (if A_LOOP is set)
302  *
303  * Then before this command, call:
304  * aSetBuffer(cmd++, 0, in, out, count)
305  *
306  * Note: count will be rounded up to the nearest multiple of 32 bytes.
307  *
308  * ADPCM decompression works on a block of 16 (uncompressed) samples.
309  * The previous 2 samples and 9 bytes of input are decompressed to
310  * 16 new samples using the code book previously loaded.
311  *
312  * Before the algorithm starts, the previous 16 samples are loaded according to flag:
313  * A_INIT: all zeros
314  * A_LOOP: the address set by aSetLoop
315  * no flags: the DRAM address in the s parameter
316  * These 16 samples are immediately copied to the destination address.
317  *
318  * The result of "count" bytes will be written after these 16 initial samples.
319  * The last 16 samples written to the destination will also be written to
320  * the state address in DRAM.
321  */

325  \
326  _a->words.w0 = _SHIFTL(A_ADPCM, 24, 8) | _SHIFTL(f, 16, 8); \
327  _a->words.w1 = (uintptr_t)(s); \

457 /*
458  * Resamples audio.
459  * Possible flags: A_INIT, A_OUT? (not used in SM64).
460  *
461  * First call:
462  * aSetBuffer(cmd++, 0, in, out, count)
463  *
464  * This command resamples the audio using the given frequency ratio (pitch)
465  * using a filter that uses a window of 4 source samples. This can be used
466  * either for just resampling audio to be able to be played back at a different
467  * sample rate, or to change the pitch if the result is played back at
468  * the same sample rate as the input.
469  *
470  * The frequency ratio is given in UQ1.15 fixed point format.
471  * For no change in frequency, use pitch 0x8000.
472  * For 1 octave up or downsampling to (roughly) half number of samples, use pitch 0xffff.
473  * For 1 octave down or upsampling to double as many samples, use pitch 0x4000.
474  *
475  * Note: count represents the number of output samples and is rounded up to
476  * the nearest multiple of 16 bytes.
477  *
478  * The state consists of the four following source samples when the algorithm stopped as
479  * well as a fractional position, and is initialized to all zeros if A_INIT is given.
480  * Otherwise it is loaded from DRAM at address s.
481  *
482  * The algorithm starts by writing the four source samples from the state (or zero)
483  * to just before the input address given. It then creates one output sample by examining
484  * the four next source samples and then moving the source position zero or more
485  * samples forward. The first output sample (when A_INIT is given) is always 0.
486  *
487  * When "count" samples have been written, the following four source samples
488  * are written to the state in DRAM as well as a fractional position.
489  */
490 #define aResample(pkt, f, p, s) \
491 { \
492  Acmd *_a = (Acmd *)pkt; \
493  \
494  _a->words.w0 = (_SHIFTL(A_RESAMPLE, 24, 8) | _SHIFTL(f, 16, 8) |\
495  _SHIFTL(p, 0, 16)); \
496  _a->words.w1 = (uintptr_t)(s); \
497 }
